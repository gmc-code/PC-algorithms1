\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18} % Set the compatibility mode to the latest version

\newcommand{\cubic}[1]{(#1)^3 - 6.5*(#1)^2 + 10*(#1) - 1.5}
\def\cubicequ{x^3 - 6.5*x^2 + 10*x - 1.5}
\def\xa{1}
\pgfmathsetmacro{\ya}{\cubic{1}}
\def\xb{3}
\pgfmathsetmacro{\yb}{\cubic{3}}
\def\xc{2}
\pgfmathsetmacro{\yc}{\cubic{2}}

\begin{document}
\begin{tikzpicture}
\begin{axis}[
    axis lines = middle,
    xlabel = $x$,
    ylabel = {$f(x)$},
    ymin=-4,
    ymax=4,
]
\addplot [
    domain=-1:5,
    samples=100,
    color=blue,
]
{\cubicequ};
\addplot[mark=*,only marks,mark size=1pt] coordinates {(\xa,\ya) (\xc,\yc) (\xb,\yb)};
\node[label={180:{f(a)}},circle,fill,inner sep=1pt] at (axis cs:\xa,\ya) {};
\node[label={90:{f(c)}},circle,fill,inner sep=1pt] at (axis cs:\xc,\yc) {};
\node[label={270:{f(b)}},circle,fill,inner sep=1pt] at (axis cs:\xb,\yb) {};
\draw[dotted] (axis cs:\xa,\ya) -- (axis cs:\xa,0);
\draw[dotted] (axis cs:\xc,\yc) -- (axis cs:\xc,0);
\draw[dotted] (axis cs:\xb,\yb) -- (axis cs:\xb,0);
\end{axis}
\end{tikzpicture}
\end{document}
